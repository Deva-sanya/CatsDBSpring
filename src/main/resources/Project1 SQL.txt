CREATE TABLE Cat (
                      id int GENERATED BY DEFAULT AS IDENTITY UNIQUE PRIMARY KEY,
                      age int not null ,
                      name varchar,
                      fatherId int,
                      motherId int,
                      gender varchar not null ,
                      color varchar,
                      person_id int  REFERENCES Person (id));
